7. validerEmail

Vérifie email : contient @ et ., local part >3, alphanumérique, ne commence pas par chiffre, TLD 2-3 lettres.
function validerEmail($email) {
    if (!filter_var($email, FILTER_VALIDATE_EMAIL)) return false;
    list($local, $domaine) = explode('@', $email);
    if (strlen($local) <= 3) return false;
    if (!ctype_alnum($local)) return false;
    if (is_numeric($local[0])) return false;
    if (!preg_match('/\.[a-zA-Z]{2,3}$/', $domaine)) return false;
    return true;
}
function validerEmail($email) {
    $pattern = '/^[a-zA-Z][a-zA-Z0-9]{3,}@[a-zA-Z0-9.-]+\.[a-zA-Z]{2,3}$/';
    return preg_match($pattern, $email) === 1;
}

function inverserChaine($str) {

3. Somme 1 + 2 + … + $n

function sommeRecursive($n) {
    if ($n <= 1) return $n;

4. PGCD (Euclide)

function pgcd($a, $b) {
    if ($b == 0) return $a;
    return pgcd($b, $a % $b);
}

5. Somme des nombres dans un tableau multidimensionnel

function sommeMultidim($array) {
    $somme = 0;
    foreach ($array as $element) {
        if (is_array($element))
            $somme += sommeMultidim($element);
        else
            $somme += $element;
    }
    return $somme;
}

6. Puissance récursive

function puissance($nombre, $exposant) {
    if ($exposant == 0) return 1;
    return $nombre * puissance($nombre, $exposant - 1);
}

7. Palindrome

function estPalindrome($mot) {
    $len = strlen($mot);
    if ($len <= 1) return true;
    if ($mot[0] != $mot[$len-1]) return false;
    return estPalindrome(substr($mot, 1, $len-2));
}

8. Suite de Fibonacci (définition : fib(0)=1, fib(1)=2, fib(2)=5)

function fib($n) {
    if ($n == 0) return 1;
    if ($n == 1) return 2;
    if ($n == 2) return 5;
    return fib($n-1) + fib($n-2);
}

9. Suite de Lucas (L0=2, L1=1, Ln = Ln-1+Ln-2)

function lucas($n) {
    if ($n == 0) return 2;
    if ($n == 1) return 1;
    return lucas($n-1) + lucas($n-2);
}

10. Ln = Fn-1 + Fn+1 (Fibonacci classique)

function fibo($n) {
    if ($n == 0) return 0;
    if ($n == 1) return 1;
    return fibo($n-1) + fibo($n-2);
}
function Ln($n) {
    return fibo($n-1) + fibo($n+1);
}

11. Somme des chiffres d’un entier

function sommeChiffres($n) {
    $n = abs($n);
    if ($n < 10) return $n;
    return ($n % 10) + sommeChiffres((int)($n / 10));
}

13. Produit de deux matrices

function produitMatrices($m1, $m2) {
    $n = count($m1);
    $res = [];
    for ($i=0; $i<$n; $i++) {
        for ($j=0; $j<$n; $j++) {
            $res[$i][$j] = array_sum(array_map(fn($k) => $m1[$i][$k] * $m2[$k][$j], range(0,$n-1)));
        }
    }
    return $res;
}

function trierTableau($tab) {
    $n = count($tab);
    for ($i=0; $i<$n-1; $i++)
        for ($j=0; $j<$n-$i-1; $j++)
            if ($tab[$j] > $tab[$j+1]) {
                $temp = $tab[$j];
                $tab[$j] = $tab[$j+1];
                $tab[$j+1] = $temp;
            }
    return $tab;
}

6. statistiques

function statistiques($tab) {
    if (empty($tab)) return ['somme'=>0, 'moyenne'=>0, 'min'=>null, 'max'=>null, 'ecart_type'=>0];
    $somme = array_sum($tab);
    $moy = $somme / count($tab);
    $min = min($tab);
    $max = max($tab);
    $var = array_sum(array_map(fn($v) => pow($v - $moy, 2), $tab)) / count($tab);
    return ['somme'=>$somme, 'moyenne'=>$moy, 'min'=>$min, 'max'=>$max, 'ecart_type'=>sqrt($var)];
}

7. tableauEnArbre

function tableauEnArbre($elements, $id = 'id', $parent = 'parent_id') {
    $index = [];
    foreach ($elements as &$e) {
        $index[$e[$id]] = &$e;
        $e['enfants'] = [];
    }
    $arbre = [];
    foreach ($elements as &$e) {
        if ($e[$parent] === null) $arbre[] = &$e;
        else $index[$e[$parent]]['enfants'][] = &$e;
    }
    return $arbre;
}
